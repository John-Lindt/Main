#include <iostream>
using namespace std;
int strlen(string a);
string upper(string a);
int main(int argc, char argv[]) {
string test="Hello";
cout << strlen(test) << endl;
cout << upper(test);
}
int strlen(string a) {
char next=a[0];
int len=0;
while((next!='/0')&&(next!=null) {
len++;
next=a[len];
}
return 0;
}
string upper(string a) {
int len=strlen(a)+1;
for(int i=0;i<len;i++) {
	if(a[i]>96 && a[i]<123) {
		a[i]=a[i]-32;
	}	
}
}
